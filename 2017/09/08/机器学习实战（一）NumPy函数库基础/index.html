from numpy import *
random.rand(4,4)
"""
mat()函数将数组转化为矩阵
"""
randMat = mat(random.rand(4,4))
"""
.I操作实现矩阵求逆运算
"""
# randMat.I
invRandMat = randMat.I
"""
执行矩阵乘法，矩阵与逆矩阵相乘
"""
randMat*invRandMat
"""
有误差，得到误差值
"""
myEye = randMat*invRandMat
myEye - eye(4)
